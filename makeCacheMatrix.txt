makeCacheMatrix <- function(mat = numeric()) {

        set <- function(y) {

                mat <<- y

                minv <<- NULL

        }

        get <- function() mat

        setinv <- function(invrs) minv <<- invrs

        getinv <- function() minv

        list(set = set, get = get,

             setinv = setinv,

             getinv = getinv)

}
